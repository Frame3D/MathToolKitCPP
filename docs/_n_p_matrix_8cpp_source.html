274 template<typename T>
275 NPMatrix<T> &NPMatrix<T>::reduce() {
276  ul_t r = 0, k, i, j;
277  NVector<T> spin;
278  for (j = 0; j < floor(_p / 2); ++j) {
279 
280  k = maxAbsIndexCol(j, r);
281  if (abs((*this)(k, j)) > EPSILON) {
282  setRow(row(k) / (*this)(k, j), k);
283  swapRow(k, r);
284 
285  spin = row(r);
286  for (i = 0; i < _n; ++i) {
287  if (i != r)

356 NPMatrix<T> NPMatrix<T>::ndiag(const std::vector<NVector<T>> &data) {
357  auto n = (long) data.size();
358  auto middle = (n - 1) / 2;
359  auto dim = data[middle].dim();
360  NPMatrix<T> diag = NPMatrix<T>::zeros(dim);
361 
362  for (long l = -middle; l <= middle; l++) {
363  for (ul_t k = 0; k < dim - abs(l); k++) {
364  if (l < 0) {
365  diag(k - l, k) = data[l + middle](k);
366  } else {
367  diag(k, k + l) = data[l + middle](k);
368  }
369  }
370  }
371  return diag;

506  ul_t i, j, k, l;
507 
508  if (_a == nullptr) { lupUpdate(); }
509 
510  if (_a != nullptr) {
511  for (j = 0; j < _a->_n; j++) {
512  for (i = 0; i < _a->_n; i++) {

563 template<typename T>
564 void NPMatrix<T>::lupUpdate() const {
565  //Returns PA such as PA = LU where P is a row p array and A = L * U;
566  ul_t i, j, k, i_max;
567 
568  lupClear();
569 
570  _a = new NPMatrix<T>(subMatrix(_i1, _j1, _i2, _j2));
571  auto *p = new std::vector<ul_t>();
572 
573  for (i = 0; i <= _a->_n; ++i)
574  p->push_back(i); //Unit p permutation, p[i] initialized with i
575 
576  if (!_a->isUpper() || !_a->isLower()) {
577  for (i = 0; i < _a->_n; ++i) {
578  i_max = _a->col(i)(i, _a->_n - 1).maxAbsIndex() + i;
579  if (abs((*_a)(i_max, i)) > EPSILON) { //matrix is not de
580  // generate
581  if (i_max != i) {
582  j = (*p)[i];
583  (*p)[i] = (*p)[i_max];
584  (*p)[i_max] = j;
585 
586  _a->swapRow(i, i_max);
587 
588  (*p)[_a->_n]++; //counting pivots starting from a->_n (for determinant)
589  }
590  } else {
591  lupClear();
592  return;
593  }
594  for (j = i + 1; j < _a->_n; ++j) {
595  (*_a)(j, i) /= (*_a)(i, i);
596  for (k = i + 1; k < _a->_n; ++k) {
597  (*_a)(j, k) -= (*_a)(j, i) * (*_a)(i, k);
598  }
599  }
600  }
601  }
602  _perm = p;
603 }
